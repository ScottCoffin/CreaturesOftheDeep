\begin{labbox}{Study Objective}
This assay evaluates \textbf{competitive feeding behavior} among marine populations operating under constrained nutrient access.
\end{labbox}

\begin{labbox}{Experimental System}
\begin{itemize}
\item Marine populations = teams
\item Nutrient reservoirs = cups
\item Plankton slurry = beverage
\item Feeding success = successful shots
\end{itemize}
\end{labbox}

\begin{labbox}{Protocol}
\begin{enumerate}
\item Initiate feeding attempts.
\item On a successful shot, the opposing participant absorbs the beverage and the matched reservoir is removed.
\item Continue until one population has no remaining reservoirs.
\end{enumerate}
\end{labbox}

\begin{labbox}{Operational Setup}
\begin{itemize}
\item Two populations compete head-to-head.
\item Reservoirs are arranged in a triangular formation.
\item Each participant is assigned a feeding position.
\item All participants may shoot continuously.
\end{itemize}
\end{labbox}

\end{minipage}\hfill
